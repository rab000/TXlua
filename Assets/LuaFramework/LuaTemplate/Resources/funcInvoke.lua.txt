--nafio info 这里func是reauire得到的lua文件的table
--nafio info args一般是自定义的ui lua文件

local function FuncInvoke(func,args)
    if func then
        local count=args and args.Length or 0
        local argTable={}
        for index=1,count do
            argTable[index]=args[index-1]
        end

		--nafio info table.pack用于把数据等转成表
		--nafio into table.unpack用于返回表中的一些元素(形成新表)

        return func(table.unpack(argTable,1,count))
		
    end
    return nil
end

--nafio info这里猜测就是require正常返回函数指针
return {FuncInvoke=FuncInvoke}
